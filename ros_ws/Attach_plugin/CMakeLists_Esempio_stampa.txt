cmake_minimum_required(VERSION 3.10)
project(HelloModelPlugin)

find_package(gz-cmake3 REQUIRED)
gz_find_package(gz-sim8 REQUIRED)
gz_find_package(gz-plugin2 REQUIRED COMPONENTS register)

add_library(HelloModelPlugin SHARED src/Attach_plugin.cc)

set_property(TARGET HelloModelPlugin PROPERTY CXX_STANDARD 17)
target_compile_definitions(HelloModelPlugin PRIVATE GZ_PLUGIN_ENABLE_REGISTER)

target_link_libraries(HelloModelPlugin
  gz-sim8::gz-sim8
  gz-plugin2::gz-plugin2
)
